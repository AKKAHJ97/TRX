      <div class="gui-title">GUI Preview</div>
      <img id="guiImage" class="gui-image" src="" alt="Script GUI Image" />
      <div class="gui-sim" id="guiSim">
        <span style="color: gray;">Waiting for script input...</span>
      </div>
      <button class="run-btn" onclick="runScript()">Run Script</button>
    </div>

  <script>
    const scriptInput = document.getElementById("scriptInput");
    const guiSim = document.getElementById("guiSim");
    const guiImage = document.getElementById("guiImage");

    function runScript() {
      alert("Script would run here! (This is a simulation button only.)");
    }

    scriptInput.addEventListener("input", () => {
      const script = scriptInput.value;
      guiSim.innerHTML = ""; // Clear preview
      guiImage.style.display = "none";

      // Show GUI image if specific pastebin link found
      if (script.includes("pastebin.com/raw/gi3FYZuB")) {
        guiImage.src = "https://i.imgur.com/wKXMEWn.png";  // Example GUI image
        guiImage.style.display = "block";
      }

      // Extract button names from script
      const matches = script.match(/Text\\s*=\\s*"([^"]+)"/g);
      if (matches) {
        matches.forEach((line) => {
          const btnText = line.match(/"([^"]+)"/)[1];
          const button = document.createElement("button");
          button.className = "gui-button";
          button.innerText = btnText;
          guiSim.appendChild(button);
        });
      } else {
        guiSim.innerHTML = "<span style='color: gray;'>No buttons detected in script.</span>";
      }
    });
  </script>
